\section*{Введение}

Вопросы G-сходимости и усреднения систем уравнений Бельтрами изучены в работах \cite{1,6,7,8}. В данной работе
предложен новый метод построения усредненной системы, основанный на методе асимптотических разложений.

\section{Первое приближение}

Систему уравнений Бельтрами удобно рассматривать в виде одного уравнения в комплексной форме:
$\partial_{\bar{z}}u+\mu(x)\partial_z u=f$, где дифференциальные
выражения $\partial_{\bar{z}}u$, $\partial_z u$ определяются формулами
$$2\partial_{\bar{z}}u=\frac{\partial u}{\partial x_1}+i\frac{\partial u}{\partial x_2}, \qquad 2\partial_z u=\frac{\partial u}{\partial x_1}-i\frac{\partial u}{\partial x_2}.$$
$i$ --- мнимая едирица.

Пусть $u_\varepsilon$ --- решение задачи Римана-Гильберта
\begin{equation}\label{1}
   \left\{\begin{array}{l}
A_\varepsilon u_\varepsilon\equiv\partial_{\bar{z}}u_\varepsilon+\mu(\varepsilon^{-1}x)\partial_z u_\varepsilon=f\in L_2(Q), \\[3mm]
   u_\varepsilon\in W_0(Q)=\big\{u\in W_2^1(Q)\mid \text{Re\,}u_{\big|\partial Q}=0,\quad
 \int\limits_{\partial Q}\text{Im\,}u\,dx=0\big\},
\end{array}\right.
\end{equation}
где $\mu(x)=\mu(x_1,x_2)$ --- периодическая функция периода $T$ по каждой переменной, причем
\begin{equation}\label{p2}
   \mathop{\rm vrai\,sup}_{x\in \mathbb{R}^2}|\mu(x)|\leqslant k_0 <1,
\end{equation}
 $Q$ --- ограниченная односвязная область плоскости с кусочно-гладкой границей; $k_0>0$ --- константа (\textit{константа эллиптичности}).

Как известно (см., например: \cite{1,6,7,9}), задача Римана—Гильберта \ref{1} однозначно разрешима для любой правой части
$f\in L_2(Q)$.
Попытаемся написать «хорошее» приближение к этому решению,
приближение, которое учитывало бы быструю осцилляцию
коэффициента уравнения. По аналогии с некоторыми приемами нелинейной
механики в работах Бахвалова \cite{10,11,2}, Санчес — Паленсии \cite{3}, а также в многочисленных физических публикациях было
показано, что первое приближение целесообразно искать в виде

\begin{equation}\label{p}
   u_1^\varepsilon(x)=u^0(x)+\varepsilon u_1(x,y),
\end{equation}
где $y=\varepsilon^{-1}x$, а $u_1(x,y)$ --- периодическая функция по $y$.
 Нам потребуется
вычислить выражение $A_\varepsilon u_1^\varepsilon$. В связи с этим
полезно предварительно выяснить, как дифференциальный
оператор $A_\varepsilon$ действует на сложную функцию $\varphi(x,y)$.



С этой целью найдем результаты действия дифференциальных выражений $\partial_{\bar{z}}$
и $\partial_z$ на сложную функцию $\varphi(x,y)$,  $y=\varepsilon^{-1}x$.
Полные производные по $z$ и $\bar{z}$ обозначим $\frac{d}{dz}$, $\frac{d}{d\bar{z}}$
соответственно. Имеем
$$
\frac{d\varphi}{dz}=\partial_z\varphi+\varepsilon^{-1}\partial_\xi \varphi,\quad
\frac{d\varphi}{d\bar{z}}=\partial_{\bar{z}}\varphi+\varepsilon^{-1}\partial_{\bar{\xi}} \varphi,\quad
\xi=y_1+iy_2.
$$
Следовательно,
$$
A_\varepsilon=\partial_{\bar{z}}+\mu(y)\partial_z+\varepsilon^{-1}\left(\partial_{\bar{\xi}}+
\mu(y)\partial_\xi \right)\equiv A_1+\varepsilon^{-1}A_2,
$$
где $A_1=\partial_{\bar{z}}+\mu(y)\partial_z$, $A_2=\partial_{\bar{\xi}}+\mu(y)\partial_\xi$.



Значит,
$$
A_\varepsilon u_1^\varepsilon=(A_1+\varepsilon^{-1}A_2)(u^0+\varepsilon u_1)=
A_1 u^0+A_2u_1+\varepsilon A_1 u_1+\varepsilon^{-1}A_2 u^0.
$$
Так как $A_2u^0=\partial_{\bar{\xi}}u^0(x)+\mu(y)\partial_{\xi}u^0(x)=0$, имеем
\begin{equation}\label{A}
   A_\varepsilon(u_1^\varepsilon-u_\varepsilon)=A_1 u^0+A_2u_1+\varepsilon A_1 u_1-f.
\end{equation}

Приравняв  нулю коэффициент при $\varepsilon^0$, получим
$$
0=A_1u^0+A_2u_1-f=
$$
$$
=\partial_{\bar{z}}u^0(x)+\mu(y)\partial_{z}u^0(x)+
\partial_{\bar{\xi}}u_1(x,y)+\mu(y)\partial_{\xi}u_1(x,y)-f.
$$



Следовательно,
\begin{equation}\label{2}
  \partial_{\bar{\xi}}u_1(x,y)+\mu(y)\partial_{\xi}u_1(x,y)=
  -\partial_{\bar{z}}u^0(x)-\mu(y)\partial_{z}u^0(x)+f(x).
\end{equation}
Задача \eqref{2} есть периодическая задача по $y$, где $x\in Q$ --- параметр.
Для разрешимости \eqref{2} необходимо и достаточно, чтобы правая часть была
ортогональна ядру сопряженного однородного уравнения. Как известно (см. \cite{}), ядро сопряженного
оператора к оператору периодической задачи одномерно $\text{Ker\,}A^\ast=\{P=cp\mid
c\in\mathbb{C}\}\subset  L_{2\,\text{per}}(\square)$, где $L_{2\,\text{per}}(\square)$ --- пространство Лебега периодических функций. При этом среднее значение $<p>=|\square|^{-1}\int_{\square}p(x)\,dx=1$, $\square$ --- <<квадрат периодов>>.  Значит,
\begin{equation}\label{3}
   \left< (\partial_{\bar{z}}u^0(x)+\mu(y)\partial_{z}u^0(x)-f(x))\overline{p(y)}\right>_y=0,
\end{equation}
где $<...>_y$ означает, что среднее значение берется по $y$.
Из \eqref{3}, с учетом $\left<p\right>=1$, получим
\begin{equation} \label{4}
   \partial_{\bar{z}}u^0(x)+\left<\mu\bar{p}\right>\partial_{z}u^0(x)=f(x),\quad u^0\in W_0(Q).
\end{equation}
 Это и есть усредненное уравнение (см. \cite{1,8}). Усредненное уравнение для уравнения Бельтрами также
является уравнением Бельтрами  (см. \cite{1,8}).



 К нему добавлено естественное условие $u^0\in W_0(Q)$.
С учетом \eqref{4} из \eqref{2} получим
\begin{equation}\label{5}
   \partial_{\bar{\xi}}u_1(x,y)+\mu(y)\partial_{\xi}u_1(x,y)=
(\left< \mu \bar{p}\right>-\mu(y))\partial_zu^0(x).
\end{equation}
Пусть $N(y)$ --- решение периодической задачи
$$
\partial_{\bar{\xi}}N+\mu\partial_\xi N= \left< \mu \bar{p}\right>-\mu,
$$
тогда решением \eqref{5} будет
$$
u_1(x,y)=N(y)\partial_z u^0(x).
$$
Следовательно, первое приближение \eqref{p} имеет вид
\begin{equation}\label{6}
   u^1_\varepsilon(x)=u^0(x)+\varepsilon N(y) \partial_z u^0(x).
\end{equation}
Тогда соотношение \eqref{A} можно представить в виде
$$
A_{\varepsilon}(u^1_{\varepsilon}-u_{\varepsilon})=\varepsilon A_1u_1
=\varepsilon N(y)(\partial^2_{z\bar{z}} u^0(x)+\mu(y)\partial^2_{zz}u^0(x)).
$$

\section{Оценка разности между точным решением и первым приближением}

 Теперь мы займемся оценкой разности между точным решением
$u_\varepsilon$ задачи Рима\-на--Гильберта \eqref{1} и первым приближением  \eqref{6} ---
$u_\varepsilon^1$.
Пусть
$u^0$ --- решение усредненной задачи Римана--Гильберта \eqref{4}, и пусть $u^0\in C^2(\overline Q)\cap W_0(Q)$.
(Чтобы обеспечить такую гладкость $u^0$ достаточно взять правую часть $f$ задачи \eqref{4} из $W^2_p(Q)$, $p>2$).
Кроме того предположим, что коэффициент $\mu$  и область $Q$  достаточно гладки. В частности, будем считать, что $N\in C^1(\square)$.

Первое приближение \eqref{6} ---
$u_\varepsilon^1$ не принадлежит пространству $W_0(Q)$. Это вызывает определенные затруднения при оценке остатка
$u_\varepsilon-u^1_\varepsilon$, чтобы обойти их введем в рассмотрение семейство функций
\begin{equation}\label{dist}
\theta^\varepsilon(x)=\omega(\varepsilon^{-1}\rho(x,\partial Q)),
\end{equation}
где $\rho(x, \partial Q)$ --- расстояние от $x$ до $\partial Q$,
$$
\omega(t)=\begin{cases}
t, &\text{если $0\leqslant t\leqslant 1$,}\\
1, &\text{если $t>1$.}
\end{cases}
$$



Семейство функций $\theta^\varepsilon(x)$
удовлетворяют следующим условиям:
\begin{enumerate}
\item\label{11n} $0\leqslant \theta^\varepsilon\leqslant1$; $\theta^\varepsilon=1$ вне $\varepsilon$-окрестности границы $Q$;
\item\label{12n} $\varepsilon|\nabla \theta^\varepsilon|\leqslant c$ в $Q$, причем константа $c$ не зависит от $\varepsilon$;
\item $\theta^\varepsilon=0$ на границе $Q$.
\end{enumerate}
  Подправим первое приближение $u^1_\varepsilon$  с тем, чтобы оно
удовлетворяло нулевым условиям Римана -- Гильберта (см. \eqref{1}). Для этого возьмем
\begin{multline*}
   w^1_\varepsilon(x)=u^0(x)+\varepsilon\theta^\varepsilon  N(y) \partial_z u^0(x)= \\
   =u^1_\varepsilon(x)-\varepsilon(1-\theta^\varepsilon ) N(y) \partial_z u^0(x).
\end{multline*}
Оценим разность
\begin{equation}\label{11}
u^1_\varepsilon(x)-w^1_\varepsilon(x)=\varepsilon(1-\theta^\varepsilon ) N(y) \partial_z u^0(x).
\end{equation}
Имеем:



\begin{multline*}
   \partial_z\left(u^1_\varepsilon(x)-w^1_\varepsilon(x)\right)=-\varepsilon\partial_z\theta^\varepsilon N(y)
   \partial_z u^0(x) +(1-\theta^\varepsilon)\partial_\zeta N(y) \partial_z u^0(x)+\\
   +\varepsilon(1-\theta^\varepsilon ) N(y) \partial_{zz} u^0(x).
\end{multline*}
Каждое слагаемое справа здесь, согласно свойству \ref{11n} функции $\theta^\varepsilon$,  равно нулю вне $\varepsilon$ окресности границы. С учетом этого и свойства \ref{12n}   функции $\theta^\varepsilon$ легко получим
\begin{equation}\label{12}
\| \partial_z\left(u^1_\varepsilon-w^1_\varepsilon\right)\|_{L_p(Q)}\leqslant c\varepsilon^{1/p},
\end{equation}
где $c>0$ ---  постоянная, независящая от $\varepsilon$, $p>1$.

Аналогичную оценку получим и для <<производной>> $\partial_{\bar z}(u^1_\varepsilon-w^1_\varepsilon)$:
\begin{equation}\label{13}
\| \partial_{\bar z}\left(u^1_\varepsilon-w^1_\varepsilon\right)\|_{L_p(Q)}\leqslant c\varepsilon^{1/p},
\end{equation}
 Из этих оценок и
\eqref{11} получим
\begin{equation}\label{14}
\| u^1_\varepsilon-w^1_\varepsilon\|_{W^1_p(Q)}\leqslant c\varepsilon^{1/p}.
\end{equation}



Теперь найдем $L_p$-оценку $f_\varepsilon:=A_\varepsilon(u_{\varepsilon}-w^1_{\varepsilon})$. Имеем:
\begin{gather*}
f_{\varepsilon}=A_{\varepsilon}(u_{\varepsilon}-u^1_{\varepsilon})+A_{\varepsilon}(u_{\varepsilon}^1-w^1_{\varepsilon})=\\=
-\varepsilon N(y)\left(\partial^2_{z\bar{z}} u^0(x)+\mu(y)\partial^2_{zz}u^0(x)\right)+\\
+ \partial_{\bar z}\left(u^1_\varepsilon-w^1_\varepsilon\right)+\mu(y) \partial_{z}\left(u^1_\varepsilon-w^1_\varepsilon\right)
\end{gather*}
Отсюда и  $L_p$-оценок производных \eqref{12}, \eqref{13} получим
\begin{equation}\label{15}
\|f_\varepsilon\|_{L_p(Q)}=\|A_\varepsilon(u_{\varepsilon}-w^1_{\varepsilon})\|_{L_p(Q)}\leqslant c\varepsilon^{1/p}.
\end{equation}
Для задачи Римана--Гильберта:
\begin{align*}
 &\partial_{\bar z}u+\mu(x) \partial_{z}u=f\in L_p(Q),\\
&u\in W_0(Q).
\end{align*}
где $\mu$ --- произвольная измеримая функция, удовлетворяющая \eqref{p2} справедливо следующее утверждение

%\begin{propos*}
% Найдется число $p>2$ такое, что для решения задачи Рима\-на--Гильберта имеет место оценка
%\begin{equation}\label{16}
%\| u\|_{W^1_p(Q)}\leqslant c\|f\|_{L_p(Q)},
%\end{equation}
%где постоянная $c>0$  зависит только от постоянной эллиптичности $k_0$.
%\end{propos*}
\medskip


\noindent \textsc{Предложение (см.\,\cite{4}).} \textit{ Найдется число $p>2$ такое, что для решения задачи Рима\-на--Гильберта имеет место оценка
\begin{equation}\label{16}
\| u\|_{W^1_p(Q)}\leqslant c\|f\|_{L_p(Q)},
\end{equation}
где постоянная $c>0$  зависит только от постоянной эллиптичности $k_0$.}
\medskip

Заметим, что разность $u_{\varepsilon}-w^1_{\varepsilon}$  принадлежит$W_0(Q)$  и является решением задачи Римана--Гильберта: $\partial_{\bar z}u+\mu(x) \partial_{z}u=f_\varepsilon$, $u\in W_0(Q)$,  поэтому из оценок \eqref{15}, \eqref{16}  получим
\begin{equation}\label{17}
\| u_{\varepsilon}-w^1_{\varepsilon}\|_{W^1_p(Q)}\leqslant c\varepsilon^{1/p}.
\end{equation}
\textit{Из оценок  \eqref{14}, \eqref{16} получим следующую оценку разности $u_\varepsilon-u^1_\varepsilon$}




\begin{equation}\label{17}
\| u_{\varepsilon}-u^1_{\varepsilon}\|_{W^1_p(Q)}\leqslant c\varepsilon^{1/p}.
\end{equation}

Согласно  \eqref{6}   получим
          % Из оценок  \eqref{14}, \eqref{17}

\begin{equation*}
\| u_{\varepsilon}^1-u^0\|_{W^1_p(Q)}\leqslant c\varepsilon^{1/p},
\end{equation*}
где постоянная $c$  зависит только от максимума модуля $N$  и  $W_p^1$-нормы  $\partial_zu^0$.
Отсюда и из \eqref{17} получим
\begin{equation*}
\| u_{\varepsilon}-u^0\|_{W^1_p(Q)}\leqslant c\varepsilon^{1/p},
\end{equation*}
где $c$  --- постоянная незавиящая от  $\varepsilon$. Следовательно,
ввиду теоремы вложения Соболева, имеет место
\begin{theorem*}
Решение $u_\varepsilon$ задачи Римана--Гильберта \eqref{1} равномерно сходится в $\overline{Q}$   к решению задачи
 Римана--Гильберта \eqref{4} при  $\varepsilon\to 0$.
\end{theorem*}
